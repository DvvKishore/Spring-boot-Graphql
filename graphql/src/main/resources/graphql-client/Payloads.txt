Add Bus (Mutation)
curl --location 'http://localhost:8080/graphql' \
--header 'Content-Type: application/json' \
--data-raw '{
  "query": "mutation { addBus(busNumber: \"KA-01-1234\", source: \"Bangalore\", destination: \"Chennai\", totalSeats: 40, womenSeats: 5, seniorSeats: 5) { id busNumber totalSeats availableNormalSeats availableWomenSeats availableSeniorSeats totalAvailableSeats status } }"
}'

Get Bus By ID (Query)
curl --location 'http://localhost:8080/graphql' \
--header 'Content-Type: application/json' \
--data-raw '{
  "query": "{ getBus(id: \"1\") { id busNumber source { name } destination { name } totalSeats totalAvailableSeats status } }"
}'

Complex Nested Query Example
curl --location 'http://localhost:8080/graphql' \
--header 'Content-Type: application/json' \
--data-raw '{
  "query": "{ busesByRoute(source: \"Bangalore\", destination: \"Chennai\") { busNumber totalSeats totalAvailableSeats source { id name } destination { id name } status } }"
}'

curl --location 'http://localhost:8080/graphql' \
--header 'Content-Type: application/json' \
--data-raw '{
  "query": "mutation AddBus($busNumber: String!, $source: String!, $destination: String!, $totalSeats: Int!, $womenSeats: Int!, $seniorSeats: Int!) { addBus(busNumber: $busNumber, source: $source, destination: $destination, totalSeats: $totalSeats, womenSeats: $womenSeats, seniorSeats: $seniorSeats) { id busNumber } }",
  "variables": {
    "busNumber": "KA-02-5678",
    "source": "Hyderabad",
    "destination": "Pune",
    "totalSeats": 50,
    "womenSeats": 5,
    "seniorSeats": 5
  }
}'
